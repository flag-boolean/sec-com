[global]
   # === Доменные параметры ===
   workgroup = EXAMPLE
   realm = EXAMPLE.LOCAL
   security = user
   kerberos method = secrets and keytab
   dedicated keytab file = /etc/krb5.keytab

   # === ID-маппинг через SSSD (обязательно для ALD Pro) ===
   # Локальные пользователи/группы (не из домена)
   idmap config * : backend = tdb
   idmap config * : range = 2000-4999

   # Доменные пользователи/группы из ALD Pro
   idmap config EXAMPLE : backend = sss
   idmap config EXAMPLE : range = 5000-999999

   # === Оптимизация для больших доменов ===
   winbind enum users = no
   winbind enum groups = no
   template homedir = /home/%U
   template shell = /bin/bash

   # === Логирование ===
   log file = /var/log/samba/log.%m
   max log size = 50
   log level = 1

   # === Поддержка расширенных прав ===
   vfs objects = acl_xattr
   map acl inherit = yes
   store dos attributes = yes
